46  /** @brief templated method to compute/retrieve an analysis for a translation unit */
47  template <class Analysis>
48  Analysis* getAnalysis() const {
49  std::string name = Analysis::name;
50  auto it = analyses.find(name);
51  if (it == analyses.end()) {
52  // analysis does not exist yet, create instance and run it.
53  auto analysis = std::make_unique<Analysis>(Analysis::name);
54  analysis->run(*this);
55  // Check it hasn't been created by someone else, and insert if not
56  std::lock_guard<std::mutex> guard(analysisLock);
57  it = analyses.find(name);
58  if (it == analyses.end()) {
59  analyses[name] = std::move(analysis);
60  }
61  }
62  return dynamic_cast<Analysis*>(analyses[name].get());
63  }

65  /** @brief get the set of alive analyses of the translation unit */

69  result.insert(a.second.get());
70  }
71  return result;
72  }
