30 /**
31  * A bidirectional mapping between tuples and reference indices.
32  */
33 class InterpreterRecordMap {
34  /** The arity of the stored tuples */
35  int arity;
36 
37  /** The mapping from tuples to references/indices */
38  map<vector<RamDomain>, RamDomain> r2i;
39 
40  /** The mapping from indices to tuples */
41  vector<vector<RamDomain>> i2r;
42 
43 public:
44  InterpreterRecordMap(int arity) : arity(arity), i2r(1) {} // note: index 0 element left free
45 
46  /**
47  * Packs the given tuple -- and may create a new reference if necessary.
48  */
49  RamDomain pack(const RamDomain* tuple) {
50  vector<RamDomain> tmp(arity);
51  for (int i = 0; i < arity; i++) {
52  tmp[i] = tuple[i];
53  }
54 
55  RamDomain index;
56 #pragma omp critical(record_pack)
57  {
58  auto pos = r2i.find(tmp);
59  if (pos != r2i.end()) {
60  index = pos->second;
61  } else {
62 #pragma omp critical(record_unpack)
63  {
64  i2r.push_back(tmp);
65  index = i2r.size() - 1;
66  r2i[tmp] = index;
67 
68  // assert that new index is smaller than the range
69  assert(index != std::numeric_limits<RamDomain>::max());
70  }
71  }
72  }
73 
74  return index;
75  }
76 
77  /**
78  * Obtains a pointer to the tuple addressed by the given index.
79  */
80  RamDomain* unpack(RamDomain index) {
81  RamDomain* res;
82 
83 #pragma omp critical(record_unpack)
84  res = &(i2r[index][0]);
85 
86  return res;
87  }
88 };

90 /**
91  * The static access function for record maps of certain arities.
92  */
93 InterpreterRecordMap& getForArity(int arity) {
94  // the static container -- filled on demand
95  static unordered_map<int, InterpreterRecordMap> maps;
96  auto pos = maps.find(arity);
97  if (pos == maps.end()) {
98  maps.emplace(arity, arity);
99  }
100 
101  return maps.find(arity)->second;
102 }
