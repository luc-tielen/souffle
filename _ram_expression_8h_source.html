45  * @class RamAbstractOperator
46  * @brief Abstract class for an operator/functor

48 class RamAbstractOperator : public RamExpression {
49 public:
50  RamAbstractOperator(std::vector<std::unique_ptr<RamExpression>> args) : arguments(std::move(args)) {}
51 
52  /** @brief Get argument values */
53  std::vector<RamExpression*> getArguments() const {
54  return toPtrVector(arguments);
55  }
56 
57  /** @brief Get i-th argument value */
58  const RamExpression& getArgument(size_t i) const {
59  assert(i >= 0 && i < arguments.size() && "argument index out of bounds");
60  return *arguments[i];
61  }
62 
63  /** @brief Get number of arguments */
64  size_t getArgCount() const {
65  return arguments.size();
66  }
67 
68  std::vector<const RamNode*> getChildNodes() const override {
69  std::vector<const RamNode*> res;
70  for (const auto& cur : arguments) {
71  res.push_back(cur.get());
72  }
73  return res;
74  }
75 
76  void apply(const RamNodeMapper& map) override {
77  for (auto& arg : arguments) {
78  arg = map(std::move(arg));
79  }
80  }
81 
82 protected:
83  /** Arguments of user defined operator */
84  std::vector<std::unique_ptr<RamExpression>> arguments;
85 
86  bool equal(const RamNode& node) const override {
87  assert(nullptr != dynamic_cast<const RamAbstractOperator*>(&node));
88  const auto& other = static_cast<const RamAbstractOperator&>(node);
89  return equal_targets(arguments, other.arguments);
90  }
91 };

93 /**
94  * @class RamIntrinsicOperator
95  * @brief Operator that represents an intrinsic (built-in) functor
96  */
97 class RamIntrinsicOperator : public RamAbstractOperator {
98 public:
99  template <typename... Args>
100  RamIntrinsicOperator(FunctorOp op, Args... args)
101  : RamAbstractOperator({std::move(args)...}), operation(op) {}
102 
103  RamIntrinsicOperator(FunctorOp op, std::vector<std::unique_ptr<RamExpression>> args)
104  : RamAbstractOperator(std::move(args)), operation(op) {}
105 
106  void print(std::ostream& os) const override {
107  if (isInfixFunctorOp(operation)) {
108  os << "(";
109  os << join(arguments, getSymbolForFunctorOp(operation),
110  print_deref<std::unique_ptr<RamExpression>>());
111  os << ")";
112  } else {
113  os << getSymbolForFunctorOp(operation);
114  os << "(";
115  os << join(arguments, ",", print_deref<std::unique_ptr<RamExpression>>());
116  os << ")";
117  }
118  }
119 
120  /** @brief Get operator symbol */
121  FunctorOp getOperator() const {
122  return operation;
123  }
124 
125  RamIntrinsicOperator* clone() const override {
126  std::vector<std::unique_ptr<RamExpression>> argsCopy;
127  for (auto& arg : arguments) {
128  argsCopy.emplace_back(arg->clone());
129  }
130  return new RamIntrinsicOperator(operation, std::move(argsCopy));
131  }
132 
133 protected:
134  /** Operation symbol */
135  const FunctorOp operation;
136 
137  bool equal(const RamNode& node) const override {
138  const auto& other = static_cast<const RamIntrinsicOperator&>(node);
139  return RamAbstractOperator::equal(node) && getOperator() == other.getOperator();
140  }

147 class RamUserDefinedOperator : public RamAbstractOperator {

149  RamUserDefinedOperator(std::string n, std::string t, std::vector<std::unique_ptr<RamExpression>> args)

153  os << "@" << name << "_" << type << "(";
154  os << join(arguments, ",",
155  [](std::ostream& out, const std::unique_ptr<RamExpression>& arg) { out << *arg; });
156  os << ")";
157  }

160  const std::string& getName() const {
161  return name;
162  }
163 
164  /** @brief Get types of arguments */
165  const std::string& getType() const {
166  return type;
167  }
168 
169  RamUserDefinedOperator* clone() const override {
170  auto* res = new RamUserDefinedOperator(name, type, {});
171  for (auto& cur : arguments) {
172  RamExpression* arg = cur->clone();
173  res->arguments.emplace_back(arg);
174  }
175  return res;
176  }
177 
178 protected:
179  /** Name of user-defined operator */
180  const std::string name;
181 
182  /** Argument types */
183  const std::string type;
184 
185  bool equal(const RamNode& node) const override {
186  const auto& other = static_cast<const RamUserDefinedOperator&>(node);
187  return RamAbstractOperator::equal(node) && name == other.name && type == other.type;
188  }
